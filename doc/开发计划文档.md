```python
"""
SQLite 数据库初始化脚本
功能：创建所有数据表、索引、初始化基础数据
"""

import sqlite3
from pathlib import Path

# 数据库文件路径
DB_PATH = Path(__file__).parent.parent / "automarketing.db"


def create_tables(conn: sqlite3.Connection):
    """创建所有数据表"""
    cursor = conn.cursor()
    
    # 1. 平台配置表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS platforms (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            code VARCHAR(50) UNIQUE NOT NULL,
            name VARCHAR(100) NOT NULL,
            status INTEGER DEFAULT 1,
            config TEXT,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
        )
    """)
    
    # 2. 业务线配置表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS business_lines (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            platform_id INTEGER NOT NULL,
            code VARCHAR(50) NOT NULL,
            name VARCHAR(100) NOT NULL,
            status INTEGER DEFAULT 1,
            config TEXT,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (platform_id) REFERENCES platforms(id),
            UNIQUE(platform_id, code)
        )
    """)
    
    # 3. 关键词配置表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS keywords (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            business_line_id INTEGER NOT NULL,
            keyword VARCHAR(200) NOT NULL,
            status INTEGER DEFAULT 1,
            priority INTEGER DEFAULT 0,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (business_line_id) REFERENCES business_lines(id),
            UNIQUE(business_line_id, keyword)
        )
    """)
    
    # 4. AI模型配置表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS ai_models (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            provider VARCHAR(50) NOT NULL,
            model_name VARCHAR(100) NOT NULL,
            api_key_encrypted TEXT NOT NULL,
            base_url VARCHAR(500),
            max_tokens INTEGER DEFAULT 2000,
            temperature REAL DEFAULT 0.7,
            top_p REAL DEFAULT 0.9,
            extra_params TEXT,
            is_active INTEGER DEFAULT 0,
            status INTEGER DEFAULT 1,
            description VARCHAR(500),
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            UNIQUE(provider, model_name)
        )
    """)
    
    # 5. AI提示词模板表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS prompt_templates (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            business_line_id INTEGER NOT NULL,
            template_code VARCHAR(50) NOT NULL,
            name VARCHAR(100) NOT NULL,
            template_content TEXT NOT NULL,
            variables TEXT,
            version INTEGER DEFAULT 1,
            status INTEGER DEFAULT 1,
            is_active INTEGER DEFAULT 0,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (business_line_id) REFERENCES business_lines(id)
        )
    """)
    
    # 5. 触达用户表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS contacts (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            platform_id INTEGER NOT NULL,
            business_line_id INTEGER NOT NULL,
            platform_user_id VARCHAR(200) NOT NULL,
            username VARCHAR(200),
            profile_url VARCHAR(500),
            is_author INTEGER DEFAULT 0,
            contact_status VARCHAR(20) DEFAULT 'pending',
            contact_attempts INTEGER DEFAULT 0,
            last_contact_at DATETIME,
            notes TEXT,
            metadata TEXT,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (platform_id) REFERENCES platforms(id),
            FOREIGN KEY (business_line_id) REFERENCES business_lines(id),
            UNIQUE(platform_id, platform_user_id)
        )
    """)
    
    # 6. 内容数据表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS contents (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            platform_id INTEGER NOT NULL,
            business_line_id INTEGER NOT NULL,
            content_type VARCHAR(20) NOT NULL,
            content_id VARCHAR(200) NOT NULL,
            content_url VARCHAR(500) NOT NULL,
            title VARCHAR(500),
            content_text TEXT,
            author_id VARCHAR(200),
            author_name VARCHAR(200),
            engagement_stats TEXT,
            ai_analysis_result TEXT,
            source_keyword VARCHAR(200),
            scraped_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (platform_id) REFERENCES platforms(id),
            FOREIGN KEY (business_line_id) REFERENCES business_lines(id),
            UNIQUE(platform_id, content_id)
        )
    """)
    
    # 7. 任务执行记录表（增强版）
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS task_executions (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            task_name VARCHAR(200),
            task_type VARCHAR(50) NOT NULL,
            business_line_id INTEGER NOT NULL,
            status VARCHAR(20) DEFAULT 'pending',
            task_config TEXT,
            total_items INTEGER DEFAULT 0,
            success_items INTEGER DEFAULT 0,
            failed_items INTEGER DEFAULT 0,
            pending_items INTEGER DEFAULT 0,
            progress INTEGER DEFAULT 0,
            start_time DATETIME,
            end_time DATETIME,
            account_id INTEGER,
            error_message TEXT,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (business_line_id) REFERENCES business_lines(id),
            FOREIGN KEY (account_id) REFERENCES accounts(id)
        )
    """)
    
    # 8. 系统配置表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS system_configs (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            config_key VARCHAR(100) UNIQUE NOT NULL,
            config_value TEXT,
            config_type VARCHAR(20) DEFAULT 'string',
            description VARCHAR(500),
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
        )
    """)
    
    # 9. 操作日志表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS operation_logs (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            operation_type VARCHAR(50) NOT NULL,
            operator VARCHAR(100),
            target_type VARCHAR(50),
            target_id INTEGER,
            operation_detail TEXT,
            ip_address VARCHAR(50),
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP
        )
    """)
    
    # 10. 账号配置表（社交媒体账号/指纹浏览器账号）
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS accounts (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            name VARCHAR(100) NOT NULL,
            platform_id INTEGER NOT NULL,
            browser_profile_id VARCHAR(100),
            status INTEGER DEFAULT 1,
            description VARCHAR(500),
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (platform_id) REFERENCES platforms(id)
        )
    """)
    
    # 11. 任务执行日志表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS task_logs (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            task_id INTEGER NOT NULL,
            log_level VARCHAR(10) DEFAULT 'info',
            message TEXT NOT NULL,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            FOREIGN KEY (task_id) REFERENCES task_executions(id)
        )
    """)
    
    conn.commit()


def create_indexes(conn: sqlite3.Connection):
    """创建索引"""
    cursor = conn.cursor()
    
    indexes = [
        "CREATE INDEX IF NOT EXISTS idx_contacts_business_status ON contacts(business_line_id, contact_status)",
        "CREATE INDEX IF NOT EXISTS idx_contacts_last_contact ON contacts(last_contact_at)",
        "CREATE INDEX IF NOT EXISTS idx_contents_business_type ON contents(business_line_id, content_type)",
        "CREATE INDEX IF NOT EXISTS idx_contents_source_keyword ON contents(source_keyword)",
        "CREATE INDEX IF NOT EXISTS idx_task_executions_status ON task_executions(status)",
        "CREATE INDEX IF NOT EXISTS idx_task_executions_business ON task_executions(business_line_id, created_at)",
        "CREATE INDEX IF NOT EXISTS idx_keywords_business ON keywords(business_line_id)",
        "CREATE INDEX IF NOT EXISTS idx_prompts_business ON prompt_templates(business_line_id)",
        "CREATE INDEX IF NOT EXISTS idx_ai_models_provider ON ai_models(provider)",
        "CREATE INDEX IF NOT EXISTS idx_ai_models_active ON ai_models(is_active)",
        "CREATE INDEX IF NOT EXISTS idx_accounts_platform ON accounts(platform_id)",
        "CREATE INDEX IF NOT EXISTS idx_accounts_status ON accounts(status)",
        "CREATE INDEX IF NOT EXISTS idx_task_logs_task ON task_logs(task_id, created_at)",
    ]
    
    for index_sql in indexes:
        cursor.execute(index_sql)
    
    conn.commit()


def init_base_data(conn: sqlite3.Connection):
    """初始化基础数据"""
    cursor = conn.cursor()
    
    # 插入默认平台

    conn.commit()


def main():
    """主函数"""
    # 删除旧数据库（如需重新初始化）
    if DB_PATH.exists():
        print(f"数据库已存在: {DB_PATH}")
        response = input("是否删除并重新创建? (y/N): ")
        if response.lower() != 'y':
            print("取消初始化")
            return
        DB_PATH.unlink()
    
    # 创建数据库连接
    conn = sqlite3.connect(DB_PATH)
    
    try:
        print("创建数据表...")
        create_tables(conn)
        
        print("创建索引...")
        create_indexes(conn)
        
        print("初始化基础数据...")
        init_base_data(conn)
        
        print(f"数据库初始化完成: {DB_PATH}")
        
    except Exception as e:
        print(f"初始化失败: {e}")
        conn.rollback()
        raise
    finally:
        conn.close()


if __name__ == "__main__":
    main()
```

### 2.2 SQLite PRAGMA 优化配置

**文件位置**: `AutoMarketing/src/core/database.py`

```python
"""
数据库连接配置
包含 SQLite PRAGMA 优化设置
"""

from sqlalchemy import create_engine, event
from sqlalchemy.orm import sessionmaker, declarative_base
from pathlib import Path

Base = declarative_base()


def get_database_url() -> str:
    """获取数据库URL"""
    db_path = Path(__file__).parent.parent.parent / "automarketing.db"
    return f"sqlite:///{db_path}"


def configure_sqlite_connection(dbapi_connection, connection_record):
    """配置 SQLite 连接优化"""
    cursor = dbapi_connection.cursor()
    
    # 启用 WAL 模式
    cursor.execute('PRAGMA journal_mode=WAL')
    
    # 降低同步级别
    cursor.execute('PRAGMA synchronous=NORMAL')
    
    # 设置缓存大小 (64MB)
    cursor.execute('PRAGMA cache_size=-16384')
    
    # 启用内存映射 (1GB)
    cursor.execute('PRAGMA mmap_size=1073741824')
    
    # 临时表使用内存
    cursor.execute('PRAGMA temp_store=MEMORY')
    
    # 启用外键约束
    cursor.execute('PRAGMA foreign_keys=ON')
    
    cursor.close()


def get_engine():
    """创建数据库引擎"""
    engine = create_engine(
        get_database_url(),
        connect_args={"check_same_thread": False},
        echo=False
    )
    
    # 注册连接事件
    event.listen(engine, "connect", configure_sqlite_connection)
    
    return engine


# 创建会话工厂
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=get_engine())


def get_db():
    """获取数据库会话"""
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
```

### 3.3 API 路由详细开发内容

|------|------|------|-------------|
| `POST /api/tasks/scrape` | POST | 创建爬虫任务 | `{business_line_id, keywords[], content_types[], max_items}` |
| `POST /api/tasks/message` | POST | 创建私信任务 | `{business_line_id, contact_ids[], template_id, variables}` |
| `GET /api/tasks` | GET | 获取任务列表 | `?status, business_line_id, page, size` |
| `GET /api/tasks/{id}` | GET | 获取任务详情 | - |
| `POST /api/tasks/{id}/stop` | POST | 停止任务 | - |
| `DELETE /api/tasks/{id}` | DELETE | 删除任务 | - |

- 任务创建后状态为 `pending`

### 3.5 AI 引擎架构（策略模式）

#### 3.5.1 设计概述

AI 引擎采用**策略模式**（Strategy Pattern）实现多 Provider 的可插拔支持。核心思想是将不同 AI 服务商的调用逻辑封装为独立的策略类，运行时根据数据库中配置的激活模型动态选择对应的策略执行。

**架构图**:

```
┌─────────────────────────────────────────────────────────────────┐
│                       AI Engine (engine.py)                     │
│  ┌───────────────────────────────────────────────────────────┐  │
│  │  1. 从数据库读取 active_model                             │  │
│  │  2. 根据 provider 选择对应的 Strategy                    │  │
│  │  3. 调用 Strategy.generate() 执行推理                     │  │
│  └───────────────────────────────────────────────────────────┘  │
└─────────────────────────────────────────────────────────────────┘
                              │
                              ▼
┌─────────────────────────────────────────────────────────────────┐
│                   Provider Strategy Registry                    │
│  ┌─────────┐  ┌─────────┐  ┌───────────┐  ┌───────────┐        │
│  │DeepSeek │  │ OpenAI  │  │Anthropic  │  │  Google   │        │
│  │Strategy │  │Strategy │  │ Strategy  │  │  Strategy │        │
│  └────┬────┘  └────┬────┘  └─────┬─────┘  └─────┬─────┘        │
│       │            │             │              │               │
│  ┌────┴────────────┴─────────────┴──────────────┴─────┐        │
│  │              BaseProvider (base.py)                 │        │
│  │  ┌─────────────────────────────────────────────┐   │        │
│  │  │  abstract generate(prompt, params)           │   │        │
│  │  │  abstract test_connection()                  │   │        │
│  │  └─────────────────────────────────────────────┘   │        │
│  └────────────────────────────────────────────────────┘        │
└─────────────────────────────────────────────────────────────────┘
```

#### 3.5.2 Provider 基类

**文件**: `src/ai/providers/base.py`

```python
"""
AI Provider 基类 - 定义策略接口
"""

from abc import ABC, abstractmethod
from typing import Optional, Dict, Any


class BaseProvider(ABC):
    """AI Provider 抽象基类"""
    
    def __init__(self, config: Dict[str, Any]):
        """
        初始化 Provider
        
        Args:
            config: 模型配置字典
                - api_key: 解密后的 API Key
                - base_url: API 基础地址
                - model_name: 模型名称
                - max_tokens: 最大 token 数
                - temperature: 温度系数
                - top_p: top_p 参数
                - extra_params: 额外参数（JSON）
        """
        self.api_key = config.get('api_key', '')
        self.base_url = config.get('base_url', '')
        self.model_name = config.get('model_name', '')
        self.max_tokens = config.get('max_tokens', 2000)
        self.temperature = config.get('temperature', 0.7)
        self.top_p = config.get('top_p', 0.9)
        self.extra_params = config.get('extra_params', {})
    
    @abstractmethod
    async def generate(self, prompt: str, **kwargs) -> str:
        """
        生成 AI 响应
        
        Args:
            prompt: 输入提示词
            **kwargs: 额外参数
            
        Returns:
            AI 生成的响应文本
        """
        pass
    
    @abstractmethod
    async def test_connection(self) -> bool:
        """
        测试连接是否正常
        
        Returns:
            True 表示连接成功，False 表示失败
        """
        pass
    
    @property
    @abstractmethod
    def provider_name(self) -> str:
        """Provider 名称"""
        pass


class ProviderFactory:
    """Provider 工厂类 - 根据 provider 类型创建实例"""
    
    _providers = {}
    
    @classmethod
    def register(cls, provider_type: str):
        """
        注册 Provider 类
        
        Args:
            provider_type: Provider 类型标识（如 'deepseek', 'openai'）
            
        Returns:
            装饰器函数
        """
        def decorator(provider_class):
            cls._providers[provider_type] = provider_class
            return provider_class
        return decorator
    
    @classmethod
    def create(cls, provider_type: str, config: Dict[str, Any]) -> BaseProvider:
        """
        创建 Provider 实例
        
        Args:
            provider_type: Provider 类型标识
            config: 模型配置字典
            
        Returns:
            BaseProvider 实例
            
        Raises:
            ValueError: Provider 类型未注册
        """
        provider_class = cls._providers.get(provider_type)
        if not provider_class:
            raise ValueError(f"Unsupported provider: {provider_type}")
        return provider_class(config)
```

"""
DeepSeek Provider 实现
"""

import httpx
from .base import BaseProvider, ProviderFactory


@ProviderFactory.register('deepseek')
class DeepSeekProvider(BaseProvider):
    """DeepSeek Provider"""
    
    @property
    def provider_name(self) -> str:
        return 'DeepSeek'
    
    async def generate(self, prompt: str, **kwargs) -> str:
        """调用 DeepSeek API 生成响应"""
        headers = {
            'Authorization': f'Bearer {self.api_key}',
            'Content-Type': 'application/json'
        }
        
        data = {
            'model': self.model_name,
            'messages': [
                {'role': 'user', 'content': prompt}
            ],
            'max_tokens': kwargs.get('max_tokens', self.max_tokens),
            'temperature': kwargs.get('temperature', self.temperature),
            'top_p': kwargs.get('top_p', self.top_p),
            **self.extra_params
        }
        
        async with httpx.AsyncClient(timeout=30) as client:
            response = await client.post(
                f'{self.base_url}/chat/completions',
                headers=headers,
                json=data
            )
            response.raise_for_status()
            result = response.json()
            return result['choices'][0]['message']['content']
    
    async def test_connection(self) -> bool:
        """测试 DeepSeek 连接"""
        try:
            await self.generate('Hello, world!')
            return True
        except Exception:
            return False

"""

from typing import Optional, Dict, Any
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from src.api.models import AIModel
from src.core.security import decrypt_api_key
from .providers.base import BaseProvider, ProviderFactory


class AIEngine:
    """AI 引擎 - 统一入口"""
    
    _instance = None

        if cls._instance is None:

    async def _load_active_model(self, db: AsyncSession):
        """从数据库加载激活的模型"""
        result = await db.execute(
            select(AIModel).where(AIModel.is_active == 1, AIModel.status == 1)
        )
        self._active_model = result.scalar_one_or_none()
        
        if self._active_model:
            api_key = decrypt_api_key(self._active_model.api_key_encrypted)
            config = {
                'api_key': api_key,
                'base_url': self._active_model.base_url,
                'model_name': self._active_model.model_name,
                'max_tokens': self._active_model.max_tokens,
                'temperature': self._active_model.temperature / 100,
                'top_p': self._active_model.top_p / 100,
                'extra_params': self._active_model.extra_params or {}
            }
            self._active_provider = ProviderFactory.create(
                self._active_model.provider, config
            )
    
    async def generate(
        self, 
        db: AsyncSession, 
        prompt: str, 
        **kwargs
    ) -> str:
        """
        生成 AI 响应
        
        Args:
            db: 数据库会话
            prompt: 输入提示词
            **kwargs: 额外参数（覆盖默认配置）
            
        Returns:
            AI 生成的响应文本
            
        Raises:
            RuntimeError: 没有激活的模型或模型配置无效
        """
        if not self._active_provider:
            await self._load_active_model(db)
        
        if not self._active_provider:
            raise RuntimeError("No active AI model configured")
        
        return await self._active_provider.generate(prompt, **kwargs)
    
    async def test_model(
        self, 
        db: AsyncSession, 
        model_id: int, 
        test_prompt: str = "Hello, world!"
    ) -> Dict[str, Any]:
        """
        测试指定模型
        
        Args:
            db: 数据库会话
            model_id: 模型 ID
            test_prompt: 测试提示词
            
        Returns:
            测试结果字典 {'success': bool, 'response': str}
        """
        result = await db.execute(select(AIModel).where(AIModel.id == model_id))
        model = result.scalar_one_or_none()
        
        if not model:
            return {'success': False, 'response': 'Model not found'}
        
        api_key = decrypt_api_key(model.api_key_encrypted)
        config = {
            'api_key': api_key,
            'base_url': model.base_url,
            'model_name': model.model_name,
            'max_tokens': model.max_tokens,
            'temperature': model.temperature / 100,
            'top_p': model.top_p / 100,
            'extra_params': model.extra_params or {}
        }
        
        try:
            provider = ProviderFactory.create(model.provider, config)
            response = await provider.generate(test_prompt)
            return {'success': True, 'response': response}
        except Exception as e:
            return {'success': False, 'response': str(e)}
    
    @property
    def active_model_info(self) -> Optional[Dict[str, Any]]:
        """获取当前激活模型信息"""
        if not self._active_model:
            return None
        
        return {
            'id': self._active_model.id,
            'provider': self._active_model.provider,
            'provider_name': self._active_provider.provider_name if self._active_provider else '',
            'model_name': self._active_model.model_name,
            'description': self._active_model.description
        }

```python
"""
安全模块 - API Key 加密解密
"""

import os
from cryptography.fernet import Fernet
from dotenv import load_dotenv


load_dotenv()

# 从环境变量获取加密密钥
AES_KEY = os.getenv('AES_ENCRYPTION_KEY')
if not AES_KEY:
    raise RuntimeError("AES_ENCRYPTION_KEY environment variable not set")

# 确保密钥是 Fernet 格式（32 URL-safe base64-encoded bytes）
try:
    fernet = Fernet(AES_KEY)
except Exception:
    raise RuntimeError("Invalid AES_ENCRYPTION_KEY format. Generate with: openssl rand -hex 32")


def encrypt_api_key(api_key: str) -> str:
    """
    加密 API Key
    
    Args:
        api_key: 原始 API Key
        
    Returns:
        加密后的字符串
    """
    if not api_key:
        return ''
    return fernet.encrypt(api_key.encode()).decode()


def decrypt_api_key(encrypted_key: str) -> str:
    """
    解密 API Key
    
    Args:
        encrypted_key: 加密后的字符串
        
    Returns:
        解密后的原始 API Key
    """
    if not encrypted_key:
        return ''
    return fernet.decrypt(encrypted_key.encode()).decode()


def mask_api_key(api_key: str) -> str:
    """
    脱敏显示 API Key
    
    Args:
        api_key: 原始 API Key
        
    Returns:
        脱敏后的字符串（如 sk-******8x2f）
    """
    if not api_key or len(api_key) <= 8:
        return '******'
    
    prefix = api_key[:4]
    suffix = api_key[-4:]
    return f"{prefix}******{suffix}"
```

## 六、接口数据格式约定

```typescript
// 成功响应
{
  "code": 200,
  "message": "success",
  "result": {
    // 数据内容
  }
}

{
  "code": 400,
  "message": "Invalid parameter",
  "result": null
}
```

### 6.2 分页响应格式

```typescript
// 分页数据
{
  "code": 200,
  "message": "success",
  "result": {
    "data": [],           // 数据列表
    "total": 100,         // 总记录数
    "page": 1,            // 当前页码
    "size": 20,           // 每页条数
    "pages": 5            // 总页数
  }
}
```

### 6.3 错误码约定

| 错误码 | 说明 |
|--------|------|
| 200 | 成功 |
| 400 | 请求参数错误 |
| 401 | 未授权 |
| 403 | 无权限 |
| 404 | 资源不存在 |
| 500 | 服务器内部错误 |
